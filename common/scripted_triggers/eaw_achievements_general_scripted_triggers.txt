	custom_trigger_tooltip = {
		tooltip = HAS_COMPLETED_AT_LEAST_ONE_ACHIEVEMENT
		any_of = {
			array = global.custom_achievements_array
			value = v
			meta_trigger = {
				text = {
					has_country_flag = unlocked_achievements_[ENTRY]
				}
				ENTRY = "[?v]"
			}
		}
	}
}
