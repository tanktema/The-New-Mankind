raise_human_rage = {
	if = {
		limit = {
			has_idea = human_coups
		}
		swap_ideas =  {
			remove_idea = human_coups
			add_idea = human_coups_up
		}
	}
	if = {
		limit = {
			has_idea = human_coups_up
		}
		swap_ideas =  {
			remove_idea = human_coups_up
			add_idea = human_hate
		}
	}
}
acceptance_expand = {
	if = {
		limit = {
			check_variable = { acceptance_value < 95 }
		}
		add_to_variable = {
			acceptance_value = 5
		}
		
	}
	if = {
		limit = {
			check_variable = { acceptance_value > 95 }
		}
		set_temp_variable = { accept_temp_check = acceptance_value }
		set_temp_variable = { temp_hundred = 100 }
		subtract_from_temp_variable = { temp_hundred = accept_temp_check }
		set_temp_variable = { vartoadd_accept = temp_hundred }
		add_to_variable = { acceptance_value = vartoadd_accept }
	}
	custom_effect_tooltip = expand_acceptance_tt
}
