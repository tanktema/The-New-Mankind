				if = {
					limit = {
						is_human_race = yes

				}

					start_civil_war = {

					}
				}
			}
			else_if = {
				limit = {
					ROOT = {
						is_human_race = yes
					}
					FROM = {
						is_human_race = no
					}
				}
				ROOT = {
					set_temp_variable = { acceptance_value_temp3 = acceptance_value_tension }
					divide_temp_variable = { acceptance_value_temp3 = 100 }

					add_stability = -acceptance_value_penalty
				}	
			}
		}
	}
